users        (id, name, email, password_hash, role, created_at)
recipes      (id, user_id[FK], title, ingredients, instructions, cook_time, servings, image_url, created_at)
reviews      (id, recipe_id[FK], user_id[FK], rating, comment, created_at)
favorites    (id, user_id[FK], recipe_id[FK])
collections  (id, user_id[FK], name)
collection_recipes (collection_id[FK], recipe_id[FK])
follows      (follower_id[FK], following_id[FK])




--------------------------------------------------------------------------------
1. ENUMS
--------------------------------------------------------------------------------
CREATE TYPE enum_dietary_category AS ENUM ('veg', 'non_veg', 'vegan', 'eggtarian');
CREATE TYPE enum_difficulty AS ENUM ('easy', 'medium', 'hard');
CREATE TYPE enum_role AS ENUM ('user', 'admin');

--------------------------------------------------------------------------------
2. TABLES
--------------------------------------------------------------------------------

TABLE: users
  - id              : SERIAL (Primary Key)
  - name            : VARCHAR(100) NOT NULL
  - email           : VARCHAR(255) NOT NULL UNIQUE (Indexed)
  - password_hash   : VARCHAR(255) NOT NULL
  - profile_image   : VARCHAR(500) NULLABLE  -- Cloudinary URL
  - bio             : TEXT NULLABLE
  - role            : enum_role NOT NULL DEFAULT 'user'
  - is_banned       : BOOLEAN NOT NULL DEFAULT false
  - created_at      : TIMESTAMPTZ NOT NULL
  - updated_at      : TIMESTAMPTZ NOT NULL

TABLE: recipes
  - id              : SERIAL (Primary Key)
  - user_id         : INTEGER NOT NULL REFERENCES users(id) ON DELETE CASCADE
  - title           : VARCHAR(255) NOT NULL (Indexed)
  - description     : TEXT NOT NULL
  - cover_image     : VARCHAR(500) NOT NULL  -- Cloudinary URL
  - dietary_type    : enum_dietary_category NOT NULL (Indexed)
  - difficulty      : enum_difficulty NOT NULL DEFAULT 'medium'
  - prep_time_mins  : INTEGER NOT NULL
  - cook_time_mins  : INTEGER NOT NULL
  - total_time_mins : INTEGER NOT NULL  -- prep + cook, set via Sequelize hook
  - servings        : INTEGER NOT NULL DEFAULT 1
  - views_count     : INTEGER NOT NULL DEFAULT 0
  - is_removed      : BOOLEAN NOT NULL DEFAULT false   -- admin soft-delete flag
  - created_at      : TIMESTAMPTZ NOT NULL
  - updated_at      : TIMESTAMPTZ NOT NULL

TABLE: recipe_ingredients
  - id              : SERIAL (Primary Key)
  - recipe_id       : INTEGER NOT NULL REFERENCES recipes(id) ON DELETE CASCADE
  - name            : VARCHAR(150) NOT NULL (Indexed)
  - quantity        : NUMERIC(6,2) NOT NULL
  - unit            : VARCHAR(50) NOT NULL

TABLE: recipe_instructions
  - id              : SERIAL (Primary Key)
  - recipe_id       : INTEGER NOT NULL REFERENCES recipes(id) ON DELETE CASCADE
  - step_number     : INTEGER NOT NULL
  - instruction_text: TEXT NOT NULL
  - step_image      : VARCHAR(500) NULLABLE
  UNIQUE(recipe_id, step_number)

TABLE: categories
  - id              : SERIAL (Primary Key)
  - name            : VARCHAR(100) NOT NULL UNIQUE   -- "Desserts", "South Indian"
  - slug            : VARCHAR(100) NOT NULL UNIQUE

TABLE: recipe_categories
  - recipe_id       : INTEGER NOT NULL REFERENCES recipes(id) ON DELETE CASCADE
  - category_id     : INTEGER NOT NULL REFERENCES categories(id) ON DELETE CASCADE
  PRIMARY KEY (recipe_id, category_id)

TABLE: reviews
  - id              : SERIAL (Primary Key)
  - user_id         : INTEGER NOT NULL REFERENCES users(id) ON DELETE CASCADE
  - recipe_id       : INTEGER NOT NULL REFERENCES recipes(id) ON DELETE CASCADE
  - rating          : INTEGER NOT NULL CHECK (rating BETWEEN 1 AND 5)
  - comment         : TEXT NULLABLE
  - created_at      : TIMESTAMPTZ NOT NULL
  UNIQUE(user_id, recipe_id)

TABLE: favorites
  - user_id         : INTEGER NOT NULL REFERENCES users(id) ON DELETE CASCADE
  - recipe_id       : INTEGER NOT NULL REFERENCES recipes(id) ON DELETE CASCADE
  - created_at      : TIMESTAMPTZ NOT NULL
  PRIMARY KEY (user_id, recipe_id)

TABLE: collections
  - id              : SERIAL (Primary Key)
  - user_id         : INTEGER NOT NULL REFERENCES users(id) ON DELETE CASCADE
  - name            : VARCHAR(150) NOT NULL
  - description     : TEXT NULLABLE
  - created_at      : TIMESTAMPTZ NOT NULL

TABLE: collection_recipes
  - collection_id   : INTEGER NOT NULL REFERENCES collections(id) ON DELETE CASCADE
  - recipe_id       : INTEGER NOT NULL REFERENCES recipes(id) ON DELETE CASCADE
  PRIMARY KEY (collection_id, recipe_id)

TABLE: follows
  - follower_id     : INTEGER NOT NULL REFERENCES users(id) ON DELETE CASCADE
  - following_id    : INTEGER NOT NULL REFERENCES users(id) ON DELETE CASCADE
  - created_at      : TIMESTAMPTZ NOT NULL
  PRIMARY KEY (follower_id, following_id)
  CHECK (follower_id <> following_id)
